Text Editor Rect 2.1.9

Rect extension for text editor.

Provides a feature to get cursor position and selection dimension in pixels.

Usage

Browser

<script src="./text-editor/index.min.js"></script>
<script src="./text-editor.rect/index.min.js"></script>
<script>
  Object.assign(TE.prototype, TE.Rect.that);
</script>

CommonJS

const TE = require('@taufik-nurrohman/text-editor');
const {that: thatRect} = require('@taufik-nurrohman/text-editor.rect');

Object.assign(TE.prototype, thatRect);

ECMAScript

import TE from '@taufik-nurrohman/text-editor';
import {that as thatRect} from '@taufik-nurrohman/text-editor.rect';

Object.assign(TE.prototype, thatRect);
